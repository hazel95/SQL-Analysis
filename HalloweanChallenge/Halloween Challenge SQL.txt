2) SELECT subject_name,painted_year FROM portraits ORDER BY painted_year ASC LIMIT 1

3)SELECT course,poison_level,round(AVG(poison_level),2)  from meals   group by course having avg(poison_level)>7


///////////////////////////////////////////////////////////////////////////////
4)SELECT a.author_id,a.name,b.title from authors a JOIN books b ON a.author_id=b.author_id where a.death_year < 1900


///////////////////////////////////////////////////////////////////////////////
5)Select experiment_name,success_rate from experiments ORDER BY success_rate DESC limit 5


///////////////////////////////////////////////////////////////////////////////
6)Select name,cause_of_death,CASE cause_of_death 
                                  WHEN 'old age' THEN 'Peaceful'                                                       				  ELSE 'Tragic' END as peacelesness FROM tombstones;


///////////////////////////////////////////////////////////////////////////////
7)Select volume, LAG(volume,1) OVER (ORDER BY chime_time) as previous_value,CASE WHEN volume < LAG(volume,1) OVER (ORDER BY chime_time) THEN 'CALM' ELSE 'LOUDER' END AS NOISE WHERE NOISE ='LOUDER'
END AS 'noisy' FROM chimes


///////////////////////////////////////////////////////////////////////////////
8)Select a.key_id,a.lock_id,AVG(b.power_level),c.door_name from attempts a JOIN keys b on a.key_id=b.key_id JOIN locks c ON a.lock_id=c.lock_id where a.success>0  GROUP BY a.key_id,a.lock_id ORDER BY AVG(b.power_level) DESC LIMIT 1
